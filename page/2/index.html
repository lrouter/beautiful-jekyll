package test.algorithm.struct;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.util.HashMap;
import java.util.Iterator;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Stack;

import javax.swing.plaf.synth.SynthSpinnerUI;

public class TestAlgorithmSuffixExpression {

private String operators = "+-*/()";
private HashMap<String, Integer> priorityMap = new HashMap<String, Integer>();
private Queue<Object> outQueue = new LinkedList<>();

public TestAlgorithmSuffixExpression() {
    //初始化优先级Map
    priorityMap.put("+", 1);
    priorityMap.put("-", 1);
    priorityMap.put("*", 2);
    priorityMap.put("/", 2);
    priorityMap.put("(", 0);
    priorityMap.put(")", 0);


}

//如果op2的优先级低于op1的，那么就返回true
public boolean isLowerOperatorsPriority(String op1, String op2) {
    int priority1 = priorityMap.get(op1);
    int priority2 = priorityMap.get(op2);
    if(priority1 > priority2)
        return true;
    else 
        return false;
}

public boolean isOperator(String c) {
    String operators = "+-*/()";

    if(operators.indexOf(c) == -1)
        return false;
    else
        return true;
}

public void outItem(Object item){
    System.out.println("item = " + item);
    outQueue.add(item);
}

public void findLeftBracket(Stack<String> operatorStack) {
    while(!operatorStack.empty()) {
        String op = operatorStack.pop();
        if(op.equals("("))
            break;
        else
            outItem(op);
    }
}

public void findLowPriorityOperator(Stack<String> operatorStack, String op1) {
    int index = 0;
    while(!operatorStack.empty()) {
        String op2 = operatorStack.get(index++);
        if(isLowerOperatorsPriority(op1, op2)) {
            operatorStack.push(op1);
            return;
        }else
            outItem(operatorStack.pop());
    }

    operatorStack.push(op1);
}

public void popStack(Stack<String> operatorStack) {
    while(!operatorStack.empty()) {
        outItem(operatorStack.pop());
    }
}

public void translate(String input) {
    if(null == input || input.length() == 0)
        return ;

    //step 1. 如果是数字，直接输出
    //step 2. 如果是运算符，
        //2.1 如果是左括号，压入栈中；
        //2.2 如果是右括号，将栈顶的元素弹出并输出，直到栈顶是1个左括号。将左括号弹出但不输出。
        //2.3 如果是其他运算符，将栈顶的元素优先级和其相等或者比它高的，弹出并且输出。将当前运算符压入栈中。
    //step 3. 如果遍历字符串完成，则将栈中的内容全部弹出并输出。
    Stack<String> operatorStack = new Stack<String>();
    outQueue.clear();
    String digitString = "";
    for(int i = 0; i < input.length(); i++) {

        //解析出一个运算符.
        char curChar = input.charAt(i);
        String c = String.valueOf(curChar);
        System.out.println("["+i+"]" + " c = " + c);
        if(isOperator(c)) {
            if(c.equals("("))
                operatorStack.push(c);
            else if(c.equals(")")) {
                //输出前1个数字
                System.out.println("digitString = " + digitString);
                outItem(Integer.parseInt(digitString));
                digitString = "";
                findLeftBracket(operatorStack);
            } else {
                //输出前1个数字
                System.out.println("digitString = " + digitString);
                outItem(Integer.parseInt(digitString));
                digitString = "";
                findLowPriorityOperator(operatorStack, c);
            }
        } else {
            digitString += c;
        }

        System.out.println("stack = " + operatorStack);
    }
    //输出前1个数字
    outItem(Integer.parseInt(digitString));

    popStack(operatorStack);

    return ;
}

public String translateQueueToString(){
    String out = "";
    while(!outQueue.isEmpty())
        out += outQueue.poll();
    return out;
}

public static void main(String[] args) {
    BufferedReader br = new BufferedReader(new 
            InputStreamReader(System.in));

    TestAlgorithmSuffixExpression test = new TestAlgorithmSuffixExpression();

    while(true) {
        System.out.println("输入表达式, 按下 'q' 键退出。");
        String exp = "";

        try {
            exp = br.readLine();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
            break;
        }

        System.out.println("您输入了："+exp);
        if(exp.equalsIgnoreCase("quit"))
            break;
        test.translate(exp);
        String out = test.translateQueueToString();
        System.out.println("结果是： " + out);
    } 
}

}

中缀表达式

后缀表达式-逆波兰表达式

中缀表达式转换为后缀表达式

2个栈

栈s1用于临时存储运算符，此运算符在栈内遵循越往栈顶优先级越高的原则；栈s1需先放入一个优先级最低的运算符，在这里假定为’#’；

栈s2用于输入逆波兰式，为方便起见；

从中缀式的左端开始逐个读取字符x。

1. 若x是操作数，

则分析出完整的运算数（在这里为方便，用字母代替数字），将x直接压入栈s2；

2. 若x是运算符，则分情况讨论：

  1. 若x是’(‘

    则直接压入栈s1；

  2. 若x是’)’

    则将栈s1栈顶的运算符，逐个出栈，依次压入栈s2，直到遇到’(‘并将其抛弃；

  3. 若x是除’(‘和’)’外的运算符，若当前栈s1的栈顶元素为’(‘

    则将x直接压入栈s1；

  4. 若x是除’(‘和’)’外的运算符，若当前栈s1的栈顶元素不为’(‘

    则将x与栈s1的栈顶元素比较，若x的优先级大于栈s1栈顶运算符优先级，则将x直接压入栈s1。否者，将栈s1的栈顶运算符弹出，压入栈s2中，直到栈s1的栈顶运算符优先级别低于（不包括等于）x的优先级。

    或栈s1的栈顶运算符为’(‘，此时再则将x压入栈s1;

3. 在进行完(2)后

检查栈s1是否为空..若不为空，则将栈中元素依次弹出并压入栈s2中（不包括’#’）；

4. 完成上述步骤后，栈s2便为逆波兰式输出结果。

但是栈s2应做一下逆序处理，因为此时表达式的首字符位于栈底；

链接使用的命令如下，我们做了修正，将-lmylib放在最后，这样符合一般的写法。原因我们后面会介绍。

  • -L选项指定了链接时搜索库文件的路径；
  • -o选项指定了可执行文件的名字；
  • -l选项，指定了需要链接哪些库。libmylib.a写为lmylib。如果为动态库，写法一样。

链接的过程为：

建立两张table

表1 unresolved symbol table，和表2 global symbol table；引用的函数或者全局变量的符号放在表。定义的全局函数或者变量原型符号放在表2。

按顺序遍历目标文件和库文件

搜索main.o

将main.c中定义的函数和全局变量放在表2中，将引用的其他文件的函数或者变量放在表1中，就是图中的sum函数。

搜索xml.o

如果xml.o中定义的符号，存在于表1中，那么将其从表1中删除。

xml.o中定义的符号，加入到表2中。如果有重复定义，增加计数。

xml.o中引用的函数或者全局变量符号，加入到表1中。

搜索libmylib.a

在-L选项指定的路径下搜索libmylib.a文件。如果找不到该文件，则在gcc的环境变量LIBRARY_PATH指定的路径中搜索。如果还找不到，则继续去系统的内定目录内/lib，/usr/lib，/usr/local/lib（64位系统时，改为lib64）。这是当初gcc内定的。

找不到libmylib.a文件就报错。

找到了libmylib.a文件之后，就解包这个.a文件，并遍历其中的.o文件。

过程和上面的main.o和xml.o类似。如果静态库中的.o文件，不包含程序引用的符号，那么该.o文件就会被bypass。

最后

如果表1 unresolved symbol table非空，报错：符号未定义。
如果表2 global symbol table中存在重复定义的符号，报错：重复定义的符号。

没有错误，那么合并对应的代码段和数据段，生成1个可执行文件。

gcc编译链接过程

在linux上，将c源码文件转化为可以执行的可执行文件，需要借助于gcc编译工具链。

使用gcc编译工具链，经过以下过程，就可以将c源码文件转化为可执行文件：

  • 预处理。生成的文件后缀为.i。
  • 汇编。生成的文件后缀为.s。
  • 编译。生成的文件后缀为.o。
  • 链接。生成可执行文件。

一般情况下，c源码不需要再人为的经过汇编处理成汇编代码文件，汇编代码文件后缀为.s。
开发人员在编译c源码时，编译器内部会做汇编和编译，生成object文件，文件后缀为.o。

汇编代码文件，直接编译即可，会生成object文件。

符号

Compound sentences are formed from simpler sentences and express relationships among the constituent sentences. There are five types of compound sentences, viz. negations, conjunctions, disjunctions, implications, and biconditionals.

A negation consists of the negation operator ¬ and an arbitrary sentence, called the target. For example, given the sentence p, we can form the negation of p as shown below.

(¬p)

A conjunction is a sequence of sentences separated by occurrences of the ∧ operator and enclosed in parentheses, as shown below. The constituent sentences are called conjuncts. For example, we can form the conjunction of p and q as follows.

(p ∧ q)

A disjunction is a sequence of sentences separated by occurrences of the ∨ operator and enclosed in parentheses. The constituent sentences are called disjuncts. For example, we can form the disjunction of p and q as follows.

(p ∨ q)

An implication consists of a pair of sentences separated by the ⇒ operator and enclosed in parentheses. The sentence to the left of the operator is called the antecedent, and the sentence to the right is called the consequent. The implication of p and q is shown below.

(p ⇒ q)

A biconditional is a combination of an implication and a reverse implication. For example, we can express the biconditional of p and q as shown below.

(p ⇔ q)
