package test.algorithm.struct;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.util.HashMap;
import java.util.Iterator;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Stack;

import javax.swing.plaf.synth.SynthSpinnerUI;

public class TestAlgorithmSuffixExpression {

private String operators = "+-*/()";
private HashMap<String, Integer> priorityMap = new HashMap<String, Integer>();
private Queue<Object> outQueue = new LinkedList<>();

public TestAlgorithmSuffixExpression() {
    //初始化优先级Map
    priorityMap.put("+", 1);
    priorityMap.put("-", 1);
    priorityMap.put("*", 2);
    priorityMap.put("/", 2);
    priorityMap.put("(", 0);
    priorityMap.put(")", 0);


}

//如果op2的优先级低于op1的，那么就返回true
public boolean isLowerOperatorsPriority(String op1, String op2) {
    int priority1 = priorityMap.get(op1);
    int priority2 = priorityMap.get(op2);
    if(priority1 > priority2)
        return true;
    else 
        return false;
}

public boolean isOperator(String c) {
    String operators = "+-*/()";

    if(operators.indexOf(c) == -1)
        return false;
    else
        return true;
}

public void outItem(Object item){
    System.out.println("item = " + item);
    outQueue.add(item);
}

public void findLeftBracket(Stack<String> operatorStack) {
    while(!operatorStack.empty()) {
        String op = operatorStack.pop();
        if(op.equals("("))
            break;
        else
            outItem(op);
    }
}

public void findLowPriorityOperator(Stack<String> operatorStack, String op1) {
    int index = 0;
    while(!operatorStack.empty()) {
        String op2 = operatorStack.get(index++);
        if(isLowerOperatorsPriority(op1, op2)) {
            operatorStack.push(op1);
            return;
        }else
            outItem(operatorStack.pop());
    }

    operatorStack.push(op1);
}

public void popStack(Stack<String> operatorStack) {
    while(!operatorStack.empty()) {
        outItem(operatorStack.pop());
    }
}

public void translate(String input) {
    if(null == input || input.length() == 0)
        return ;

    //step 1. 如果是数字，直接输出
    //step 2. 如果是运算符，
        //2.1 如果是左括号，压入栈中；
        //2.2 如果是右括号，将栈顶的元素弹出并输出，直到栈顶是1个左括号。将左括号弹出但不输出。
        //2.3 如果是其他运算符，将栈顶的元素优先级和其相等或者比它高的，弹出并且输出。将当前运算符压入栈中。
    //step 3. 如果遍历字符串完成，则将栈中的内容全部弹出并输出。
    Stack<String> operatorStack = new Stack<String>();
    outQueue.clear();
    String digitString = "";
    for(int i = 0; i < input.length(); i++) {

        //解析出一个运算符.
        char curChar = input.charAt(i);
        String c = String.valueOf(curChar);
        System.out.println("["+i+"]" + " c = " + c);
        if(isOperator(c)) {
            if(c.equals("("))
                operatorStack.push(c);
            else if(c.equals(")")) {
                //输出前1个数字
                System.out.println("digitString = " + digitString);
                outItem(Integer.parseInt(digitString));
                digitString = "";
                findLeftBracket(operatorStack);
            } else {
                //输出前1个数字
                System.out.println("digitString = " + digitString);
                outItem(Integer.parseInt(digitString));
                digitString = "";
                findLowPriorityOperator(operatorStack, c);
            }
        } else {
            digitString += c;
        }

        System.out.println("stack = " + operatorStack);
    }
    //输出前1个数字
    outItem(Integer.parseInt(digitString));

    popStack(operatorStack);

    return ;
}

public String translateQueueToString(){
    String out = "";
    while(!outQueue.isEmpty())
        out += outQueue.poll();
    return out;
}

public static void main(String[] args) {
    BufferedReader br = new BufferedReader(new 
            InputStreamReader(System.in));

    TestAlgorithmSuffixExpression test = new TestAlgorithmSuffixExpression();

    while(true) {
        System.out.println("输入表达式, 按下 'q' 键退出。");
        String exp = "";

        try {
            exp = br.readLine();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
            break;
        }

        System.out.println("您输入了："+exp);
        if(exp.equalsIgnoreCase("quit"))
            break;
        test.translate(exp);
        String out = test.translateQueueToString();
        System.out.println("结果是： " + out);
    } 
}

}
